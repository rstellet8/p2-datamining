Multinominal Naive Bayes¶

Uma das hipteses fundamentais desse modelo que as variveis independentes so distribudas normal e independentemente.

Entretanto, esperado que os dados aqui usados no sigam essa hiptese.

Por exemplo, esperado que as temperaturas influenciem umas as outras.

Alm disso, Esse modelo no aceitou os dados negativos de temperatura, portanto os dados sero normalizados para o intervalo [0, 1].

In [14]:
from sklearn.naive_bayes import MultinomialNB


mnbData = data_reduced.copy()
ColToNormalize = ["MinTemp", "MaxTemp", "Temp9am", "Temp3pm", "TempDiff", "HumidityDiff"]
mnbData[ColToNormalize] = minmax_scale(mnbData[ColToNormalize])

class ModeloMNB:
    def __init__(self, dados):

        x_train, x_test, y_train, y_test = train_test_split(
            mnbData.drop(columns="RainTomorrow"),
            mnbData["RainTomorrow"],
            random_state=0
        )

        MultiNB = MultinomialNB().fit(x_train, y_train)
        y_pred = MultiNB.predict(x_test)

        self.CM = confusion_matrix(y_test, y_pred)
        self.Scores = precision_recall_fscore_support(y_test, y_pred)
        self.Report = classification_report(y_test, y_pred)

mnb1 = ModeloMNB(mnbData)

from sklearn.linear_model import LogisticRegression


class ModeloRegressaoLogistica:
    def __init__(self, dados):
        logRegData = dados.copy()
        logRegData.apply(minmax_scale) # O algoritmo no converge sem essa normalizao e sem aumentar o nmero de iteraes

        x_train, x_test, y_train, y_test = train_test_split(
            logRegData.drop(columns="RainTomorrow"),
            logRegData["RainTomorrow"],
            random_state=0
        )
 
        LogReg = LogisticRegression(max_iter=1500).fit(x_train, y_train)
        y_pred = LogReg.predict(x_test)

        self.CM = confusion_matrix(y_test, y_pred)
        self.Scores = precision_recall_fscore_support(y_test, y_pred)
        self.Report = classification_report(y_test, y_pred)

def montarTabela(mnbScore, reglogScore, rf2Score):
    tabela = pd.Series([mnbScore, reglogScore, rf2Score]).to_list()

    precision = [tabela[i][0] for i in range(3)]
    recall = [tabela[i][1] for i in range(3)]
    fscore = [tabela[i][0] for i in range(3)]
    support = [tabela[i][0] for i in range(3)]

    return pd.DataFrame({
        "Precision|0": [round(i[0], 3) for i in precision],
        "Precision|1": [round(i[1], 3) for i in precision],
        "Recall|0": [round(i[0], 3) for i in recall],
        "Recall|1": [round(i[1], 3) for i in recall],
        "F-score|0": [round(i[0], 3) for i in fscore],
        "F-score|1": [round(i[1], 3) for i in fscore],
        "Support|0": [round(i[0], 3) for i in support],
        "Support|1": [round(i[1], 3) for i in support],
    }, index=["Multinomial NB", "Regresso Logstica", "Random Forest"])

Rebalanceamento dos Dados¶

A varivel dependente desbalanceada, o que pode prejudicar a qualidade de alguns modelos, como a regresso logstica.

Para corrigir esse problema usaremos as ferramentas de oversampling do pacote imblearn.
